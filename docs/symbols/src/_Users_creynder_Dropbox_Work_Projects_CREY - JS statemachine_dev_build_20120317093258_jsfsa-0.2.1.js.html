 38     var Dispatcher = function(){
 39         this._listeners = {};

 42     /**
 43      * Registers a <code>handler</code> function to be called when an event with <code>eventName</code> is dispatched.
 44      * The <code>handler</code> is called within the scope of the dispatcher.
 45      * @param {String} eventName
 46      * @param {Function} handler
 47      * @return {Object} the dispatcher
 48      */
 49     Dispatcher.prototype.addListener = function( eventName, handler ){
 50         if( ! this._listeners.hasOwnProperty( eventName ) ){
 51             this._listeners[ eventName ] = [];
 52         }
 53         this._listeners[ eventName ].push( handler );
 54 
 55         return this;
 56     };
 57 
 58     /**
 59      * Registers several <code>handlers</code> for an event with name <code>eventName</code>
 60      * @param {String} eventName
 61      * @param {Function} handlers
 62      * @return {Object} the dispatcher
 63      */
 64     Dispatcher.prototype.addListeners = function( eventName, handlers ){
 65         if( typeof handlers === "function" ){
 66             return this.addListener( eventName, handlers );
 67         }
 68 
 69         var arr;
 70         if( ! this._listeners.hasOwnProperty( eventName ) ){
 71             arr = [];
 72         }else{
 73             arr = this._listeners[ eventName ];
 74         }
 75 
 76         this._listeners[ eventName ] = arr.concat( handlers );
 77 
 78         return this;
 79     };
 80 
 81     /**
 82      * Checks whether the dispatcher has a specific <code>handler</code> registered for <code>eventName</code>
 83      * @param {String} eventName
 84      * @param {Function} handler
 85      * @return {Boolean}
 86      */
 87     Dispatcher.prototype.hasListener = function( eventName, handler ){
 88         if( this._listeners.hasOwnProperty( eventName ) ){
 89             var index = this._listeners[ eventName ].indexOf( handler );
 90             return index >= 0;
 91         }
 92         return false;
 93     };
 94     /**
 95      * Unregisters a <code>handler</code> for <code>eventName</code>
 96      * @param {String} eventName
 97      * @param {Function} handler
 98      * @return {Object} the dispatcher
 99      */
100     Dispatcher.prototype.removeListener = function( eventName, handler ){
101         if( this._listeners.hasOwnProperty( eventName ) ){
102             var index = this._listeners[ eventName ].indexOf( handler );
103             if( index >= 0 ){
104                 this._listeners[ eventName ].splice( index, 1 );
105             }
106         }
107         return this;
108     };
109 
110     /**
111      * Dispatches an event.
112      * @param {Object} e
113      * @param {String} e.type
114      * @return {Boolean} the result of the registered handlers
115      */
116     Dispatcher.prototype.dispatch = function( e ){
117         var eventName = e.type;
118         var result = true;
119         if( this._listeners.hasOwnProperty( eventName ) ){
120             var args = Array.prototype.slice.call( arguments );
121             for( var i=0, n=this._listeners[ eventName ].length ; i<n ; i++ ){
122                 var handler = this._listeners[ eventName ][ i ];
123                 result = handler.apply( this, args ) && result;
124             }
125         }
126         return result;
127     };

439     /**
440      * Parses the <code>data</code> to configure the state.
441      * @example //Strict syntax
442     var guard = function( e ){
443         return false;
444     };
445     var listener = function( e ){
446         console.log( e );
447     }
448     var config = {
449         isInitial : true,
450         transitions : {
451             'ignite' : 'on', //transitionName : targetStateName
452             'fail' : 'broken'
453         },
454         guards : {
455             entry : [ guard ],
456             exit : [ guard ]
457         },
458         listeners : {
459             entryDenied : [ listener ],
460             entered : [ listener ],
461             exitDenied : [ listener ],
462             exited : [ listener ]
463         }
464     }
465      * @example //Loose syntax
466     var guard = function( e ){
467         return false;
468     };
469     var listener = function( e ){
470         console.log( e );
471     }
472     var config = {
473         isInitial : true,
474         'ignite' : 'on', //transitionName : targetStateName
475         'fail' : 'broken'
476         guards : {
477             entry : guard,
478             exit : guard
479         },
480         listeners : {
481             entryDenied : listener,
482             entered : listener,
483             exitDenied : listener,
484             exited : listener
485         }
486     }
487      * @param {Object} data
488      */
489     jsfsa.State.prototype.parseData = function( data ){
490         if( data ){
491             if( data.isInitial ) {
492                 this.isInitial = true;
493             }
494             if( data.guards ){
495                 if( data.guards[ jsfsa.Action.ENTRY ] ){
496                     this.addGuards( jsfsa.Action.ENTRY, data.guards[ jsfsa.Action.ENTRY ] );
497                 }
498                 if( data.guards[ jsfsa.Action.EXIT ] ){
499                     this.addGuards(jsfsa.Action.EXIT, data.guards[ jsfsa.Action.EXIT ] );
500                 }
501             }
502             for( var eventName in data.listeners ){
503                 if( data.listeners.hasOwnProperty( eventName ) ){
504                     this.addListeners( eventName, data.listeners[ eventName ] );
505                 }
506             }
507             if( data.parent ){
508                 this.parent = data.parent;
509             }
510             if( data.transitions ){
511                 this._addTransitions( data.transitions );
512             }
513 
514             this._addTransitions( data, jsfsa.State._configMembers );
515         }
516     };

785     /**
786      * Accepts any number of arguments after <code>transitionName</code> that will be passed on to the
787      * guards and actions
788      * @param {String} transitionName
789      * @return {jsfsa.Automaton} the instance of {@link jsfsa.Automaton} that is acted upon
790      */
791     jsfsa.Automaton.prototype.doTransition = function( transitionName ){
792         var runner;
793         var found = false;
794         for( var i=this._currentBranch.length -1, n = 0  ; i>=n ; i-- ){
795             runner = this._currentBranch[ i ].state;
796             if( runner.hasTransition( transitionName ) ){
797                 found = true;
798                 break;
799             }
800         }
801 
802         var payload;
803         if( arguments.length > 1 ){
804             payload =  Array.prototype.slice.call( arguments );
805             payload.shift(); //drop transitionname
806         }else{
807             payload = [];
808         }
809 
810         var event = new jsfsa.StateEvent( '', this.getCurrentState().name, undefined, transitionName );
811         var args = cloneAndUnshift( payload, event.clone()._setType(jsfsa.StateEvent.TRANSITION_DENIED) );
812 
813         if( ! found ){
814             //there's no transition with that name in the current state branch
815             this.dispatch.apply( this, args );
816         }else{
817             //transition found somewhere in the _currentStateBranch
818             var targetNode = this._nodes[ runner.getTransition( transitionName ) ];
819             if( ! targetNode ){
820                 //state doesn't exist
821                 this.dispatch.apply( this, args );
822             }else{
823                 event.to = targetNode.state.name;
824                 var initialNodes = targetNode.getInitialBranch();
825                 this._newBranch = this._getBranchFromRoot( targetNode ).concat( initialNodes );
826                 var streams = this._getShortestRoute( this._currentBranch, this._newBranch );
827                 this._internalState = 'guarding';
828                 args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.Action.EXIT ) );
829                 var proceed = this._executeGuards( streams.up, args  );
830                 if( !proceed ){
831                     args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.StateEvent.EXIT_DENIED ) );
832                     this.dispatch.apply( this, args );
833                 }else{
834                     args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.Action.ENTRY ) );
835                     proceed = this._executeGuards( streams.down, args );
836                 }
837 
838                 if( ! proceed ){
839                     args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.StateEvent.ENTRY_DENIED ) );
840                     this.dispatch.apply( this, args );
841                     this._internalState = 'ready';
842                 }else{
843                     this._internalState = 'transitioning';
844                     var referer = [ { state : this } ];
845                     args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.StateEvent.EXITED ) );
846                     this._addToQueue( streams.up, args );
847                     this._addToQueue( referer, args );
848                     args = cloneAndUnshift( payload, event.clone()._setType( jsfsa.StateEvent.ENTERED ) );
849                     this._addToQueue( streams.down, args );
850                     this._addToQueue( referer, args );
851                     this.proceed();
852                 }
853             }
854         }
855 
856         return this;
857     };

869             }
870         }

875     /**
876      * @return {jsfsa.Automaton} the instance of {@link jsfsa.Automaton} that is acted upon
877      */
878     jsfsa.Automaton.prototype.proceed = function(){
879         if( this._internalState !== 'ready' && this._internalState !== 'guarding' ){
880             if( this._queue.length > 0 ){
881                 this._internalState = 'transitioning';
882                 var o = this._queue.shift();
883                 var state = o.obj.state;
884                 state._executeAction( o.args );
885                 if( this._internalState !== "paused" ){
886                     this.proceed();
887                 }
888             }else{
889                 this._finishTransition();
890             }
891         }
892 
893         return this;
894     };

917     /**
918      * @private
919      * @param {Node} node
920      * @return {Node[]}
921      */
922     jsfsa.Automaton.prototype._getBranchFromRoot = function( node ){
923         var branch = [];
924         while( node ){
925             branch.unshift( node );
926             node = node.parent;
927         }

932     /**
933      * @private
934      * @param {Node[]} rootToBegin
935      * @param {Node[]} rootToEnd
936      * @return {Object}
937      */
938     jsfsa.Automaton.prototype._getShortestRoute = function( rootToBegin, rootToEnd ){
939         var i, n = Math.min( rootToBegin.length, rootToEnd.length );
940         for( i=0 ; i<n ; i++ ){
941             if( rootToBegin[ i ] !== rootToEnd[ i ] ){
942                 break;
943             }
944         }
945 
946         var up = rootToBegin.slice( i ).reverse();
947         var down = rootToEnd.slice( i );
948 
949         return {
950             up : up,
951             down : down
952         };
953     };

955     /**
956      * @private
957      * @param {Node[]} nodesList
958      * @param {Array} args
959      */
960     jsfsa.Automaton.prototype._executeGuards = function( nodesList, args ){
961         var result = true;
962         for( var i=0, n=nodesList.length; i<n ; i++){
963             var state = nodesList[ i ].state;
964             result = state._executeGuards( args ) && result;
965         }
966 
967         return result;
968     };

972             this._queue.push( { obj : list[ i ], args : args } );
