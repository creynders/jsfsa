 44      * @param {String} eventName
 45      * @param {Function} handler
 46      */
 47     Dispatcher.prototype.addListener = function( eventName, handler ){
 48         if( ! this._listeners.hasOwnProperty( eventName ) ){
 49             this._listeners[ eventName ] = [];
 50         }
 51         this._listeners[ eventName ].push( handler );
 52 
 53         return this;
 54     };
 55 
 56     /**
 57      *
 58      * @param {String} eventName
 59      * @param {Function} handlers
 60      */
 61     Dispatcher.prototype.addListeners = function( eventName, handlers ){
 62         if( typeof handlers === "function" ){
 63             return this.addListener( eventName, handlers );
 64         }
 65 
 66         var arr;
 67         if( ! this._listeners.hasOwnProperty( eventName ) ){
 68             arr = [];
 69         }else{
 70             arr = this._listeners[ eventName ];
 71         }
 72 
 73         this._listeners[ eventName ] = arr.concat( handlers );
 74 
 75         return this;
 76     };
 77 
 78     /**
 79      *
 80      * @param {String} eventName
 81      * @param {Function} handlers
 82      * @return {Boolean}
 83      */
 84     Dispatcher.prototype.hasListener = function( eventName, handler ){
 85         if( this._listeners.hasOwnProperty( eventName ) ){
 86             var index = this._listeners[ eventName ].indexOf( handler );
 87             return index >= 0;
 88         }
 89         return false;
 90     };
 91     /**
 92      *
 93      * @param {String} eventName
 94      * @param {Function} handlers
 95      */
 96     Dispatcher.prototype.removeListener = function( eventName, handler ){
 97         if( this._listeners.hasOwnProperty( eventName ) ){
 98             var index = this._listeners[ eventName ].indexOf( handler );
 99             if( index >= 0 ){
100                 this._listeners[ eventName ].splice( index, 1 );
101             }
102         }
103         return this;
104     };

106     /**
107      *
108      * @param {Object} e
109      * @param {String} e.type
110      */
111     Dispatcher.prototype.dispatch = function( e ){
112         var eventName = e.type;
113         var result = true;
114         if( this._listeners.hasOwnProperty( eventName ) ){
115             var args = Array.prototype.slice.call( arguments );
116             for( var i=0, n=this._listeners[ eventName ].length ; i<n ; i++ ){
117                 var handler = this._listeners[ eventName ][ i ];
118                 result = handler.apply( this, args ) && result;
119             }
120         }
121         return result;
122     };

673     /**
674      * Accepts any number of arguments after <code>transitionName</code> that will be passed on to the
675      * guards and actions
676      * @param {String} transitionName
677      * @return {fsa.Automaton} the instance of {@link fsa.Automaton} that is acted upon
678      */
679     fsa.Automaton.prototype.doTransition = function( transitionName ){
680         var runner;
681         var found = false;
682         for( var i=this._currentBranch.length -1, n = 0  ; i>=n ; i-- ){
683             runner = this._currentBranch[ i ].state;
684             if( runner.hasTransition( transitionName ) ){
685                 found = true;
686                 break;
687             }
688         }
689         if( found ){
690             //transition found somewhere in the _currentStateBranch
691             var targetNode = this._nodes[ runner.getTransition( transitionName ) ];
692             if( targetNode ){
693                 //TODO: determine what to do if targetNode not found?? Currently failing silenlty
694 
695                 var initialNodes = targetNode.getInitialBranch();
696                 this._newBranch = this._getBranchFromRoot( targetNode ).concat( initialNodes );
697                 var streams = this._getShortestRoute( this._currentBranch, this._newBranch );
698                 var currentStateName = this.getCurrentState().name;
699                 var newStateName = targetNode.state.name;
700 
701                 var payload;
702                 if( arguments.length > 1 ){
703                     payload =  Array.prototype.slice.call( arguments );
704                     payload.shift(); //drop transitionname
705                 }else{
706                     payload = [];
707                 }
708 
709                 this._internalState = 'guarding';
710 
711                 var args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.Action.EXIT, currentStateName, newStateName ) );
712                 var proceed = this._executeGuards( streams.up, args  );
713                 if( !proceed ){
714                     args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.StateEvent.EXIT_DENIED, currentStateName, newStateName ) );
715                     this.dispatch.apply( this, args );
716                 }else{
717                     args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.Action.ENTRY, currentStateName, newStateName ) );
718                     proceed = this._executeGuards( streams.down, args );
719                 }
720 
721                 if( ! proceed ){
722                     args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.StateEvent.ENTRY_DENIED, currentStateName, newStateName ) );
723                     this.dispatch.apply( this, args );
724                     this._internalState = 'ready';
725                 }else{
726                     this._internalState = 'transitioning';
727                     var referer = [ { state : this } ];
728                     args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.StateEvent.EXITED, currentStateName, newStateName ) );
729                     this._addToQueue( streams.up, args );
730                     this._addToQueue( referer, args );
731                     args = cloneAndUnshift( payload, new fsa.StateEvent( fsa.StateEvent.ENTERED, currentStateName, newStateName ) );
732                     this._addToQueue( streams.down, args );
733                     this._addToQueue( referer, args );
734                     this.proceed();
735                 }
736             }
737         }
738     };

812     fsa.Automaton.prototype._getShortestRoute = function( rootToBegin, rootToEnd ){
813         var i, n = Math.min( rootToBegin.length, rootToEnd.length );
814         for( i=0 ; i<n ; i++ ){
815             if( rootToBegin[ i ] !== rootToEnd[ i ] ){
816                 break;

824             up : up,
825             down : down
