  6 describe("fsa.Automaton", function(){
  7     var sm;
  8     beforeEach( function(){
  9         sm = new fsa.Automaton();
 10     });
 11     beforeEach( function(){
 12         this.addMatchers({
 13             toBeInstanceOf : function( expected ){
 14                 return this.actual instanceof expected;
 15             }
 16         })
 17     } );
 18     afterEach( function(){
 19         sm.destroy();
 20         sm = undefined;
 21     });
 22 
 23     describe("instance", function(){
 24         it( "should be of type fsa.Automaton", function(){
 25             expect( sm ).toBeInstanceOf( fsa.Automaton );
 26         });
 27         it( "should have a root state", function(){
 28             expect( sm.getRootState() ).toBeInstanceOf( fsa.State );
 29         })
 30     });
 31 
 32     describe( "an added state", function(){
 33         it( "should be detectable", function (){
 34             var state = new fsa.State( 'foo' );
 35             sm.addState( state );
 36             expect( sm.hasState( state.name ) ).toBeTruthy();
 37         } );
 38         it( "should be removable", function(){
 39             var state = new fsa.State( 'foo' );
 40             sm
 41                 .addState( state )
 42                 .removeState( state.name )
 43             ;
 44             expect( sm.hasState( state.name ) ).toBeFalsy();
 45         });
 46         it( "should be retrievable", function(){
 47             var state = new fsa.State( 'foo' );
 48             sm.addState( state );
 49             expect( sm.getState( state.name ) ).toEqual( state );
 50         });
 51     });
 52 
 53     describe( "in simple usage the automaton", function(){
 54         beforeEach( function(){
 55             sm.createState( 'off', { transitions : { 'ignite' : 'on' }, isInitial : true } );
 56             sm.createState( 'on', { transitions : { 'shutdown' : 'off' } } );
 57         });
 58         it( "should be able to transition to another state", function(){
 59             sm.doTransition( 'ignite' );
 60             expect( sm.getCurrentState() ).toEqual( sm.getState( 'on' ) );
 61         } );
 62         it( "should be able to transition back", function(){
 63             sm.doTransition( 'ignite' );
 64             sm.doTransition( 'shutdown' );
 65             expect( sm.getCurrentState() ).toEqual( sm.getState( 'off' ) );
 66         } );
 67         it( "should fail silently if transition is not allowed", function(){
 68             sm.doTransition( 'ignite' );
 69             sm.doTransition( 'ignite' );
 70             expect( sm.getCurrentState() ).toEqual( sm.getState( 'on' ) );
 71         });
 72         it( "should fail silently if transition is not found", function(){
 73             sm.doTransition( 'ignite' );
 74             sm.doTransition( 'whatever' );
 75             expect( sm.getCurrentState() ).toEqual( sm.getState( 'on' ) );
 76         });
 77         it( "should fail silently if target is not found", function(){
 78             sm.doTransition( 'ignite' );
 79             sm.removeState( 'off' );
 80             sm.doTransition( 'shutdown' );
 81             expect( sm.getCurrentState() ).toEqual( sm.getState( 'on' ) );
 82         });
 83         it( "should fail silently if no states present", function(){
 84             sm.doTransition( 'ignite' );
 85             sm.removeState( 'off' );
 86             sm.removeState( 'on' );
 87             sm.doTransition( 'shutdown' );
 88         });
 89         it( "should revert to the parent node if a current state is removed", function(){
 90             sm.doTransition( 'ignite' );
 91             sm.removeState( 'on' );
 92             expect( sm.getCurrentState() ).toEqual( sm.getRootState() );
 93         } );
 94     });
 95     describe( "the automaton", function(){
 96         beforeEach( function(){
 97             sm.createState( 'green',    { transitions : { 'next' : 'orange' }, isInitial : true } );
 98             sm.createState( 'orange',   { transitions : { 'next' : 'red'    } } );
 99             sm.createState( 'red',      { transitions : { 'next' : 'green'  } } );
100         } );
101         it( "should allow for multiple transitions with the same name", function(){
102             sm.doTransition( 'next' );
103             expect( sm.getCurrentState() ).toEqual( sm.getState( 'orange' ) );
104             sm.doTransition( 'next' );
105             expect( sm.getCurrentState() ).toEqual( sm.getState( 'red' ) );
106             sm.doTransition( 'next' );
107             expect( sm.getCurrentState() ).toEqual( sm.getState( 'green' ) );
108         });
109         it( "should terminate transition with guards denying entry", function(){
110             var orange = sm.getState( 'orange' );
111             orange.addGuard( 'enter', function( ){return false } );
112             sm.doTransition( 'next' );
113             sm.doTransition( 'next' );
114             expect( sm.getCurrentState() ).toEqual( sm.getState( 'green' ) );
115         });
116         it( "should terminate transition with guards denying exit", function(){
117             var orange = sm.getState( 'orange' );
118             orange.addGuard( 'exit', function( ){return false } );
119             sm.doTransition( 'next' );
120             sm.doTransition( 'next' );
121             expect( sm.getCurrentState() ).toEqual( sm.getState( 'orange' ) );
122         });
123 
124     });
125     describe( "the automaton", function(){
126         it( "should be configurable through an object", function(){
127             var config = {
128                 "off" : {
129                     isInitial : true,
130                     "powerOn" : "on"
131                 },
132                 "off/standby" : {
133                     isInitial : true
134                 },
135                 "off/kaput" : {},
136                 "off/kaput/fixable" :{
137                     isInitial : true,
138                     "fixed" : "off/standby"
139                 },
140                 "off/kaput/pertetotale":{},
141                 "on" : {
142                     "powerOff" : "off",
143                     "fail" : "off/kaput",
144                     "vandalize" : "off/kaput/pertetotale"
145                 },
146                 "on/green" : {
147                     isInitial : true,
148                     "next" : "on/orange"
149                 },
150                 "on/orange" : {
151                     "next" : "on/red"
152                 },
153                 "on/red" : {
154                     "next" : "on/green"
155                 }
156             };
157             var fsm = new fsa.Automaton( config );
158             expect( fsm.getCurrentState() ).toEqual( fsm.getState('off/standby' ) );
159             fsm.doTransition( 'powerOn' );
160             expect( fsm.getCurrentState() ).toEqual( fsm.getState('on/green' ) );
161             fsm.doTransition( 'next' );
162             expect( fsm.getCurrentState() ).toEqual( fsm.getState('on/orange' ) );
163             fsm.doTransition( 'fail' );
164             expect( fsm.getCurrentState() ).toEqual( fsm.getState('off/kaput/fixable' ) );
165             fsm.doTransition( 'fixed' );
166             expect( fsm.getCurrentState() ).toEqual( fsm.getState('off/standby' ) );
167             fsm.doTransition( 'powerOn' );
168             expect( fsm.getCurrentState() ).toEqual( fsm.getState('on/green' ) );
169             fsm.doTransition( 'vandalize' );
170             expect( fsm.getCurrentState() ).toEqual( fsm.getState('off/kaput/pertetotale' ) );
171         })
172     })
173 
174 });
