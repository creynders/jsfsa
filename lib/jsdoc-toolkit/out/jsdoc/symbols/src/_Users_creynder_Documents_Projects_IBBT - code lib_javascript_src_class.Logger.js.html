 66 /**
 67  * ALogger
 68  * -------
 69  * Abstract class
 70  *
 71  * @class ALogger
 72  * @author camille.reynders@ibbt.be
 73  * @version 1.0.0
 74  * @constructor
 75  */
 76 function ALogger(){
 77 }
 78 
 79 //-- public instance props
 80 
 81 ALogger.prototype.name = 'ALogger';
 82 ALogger.prototype.enabled = true;
 83 ALogger.prototype.levelTreshold = LogLevel.DEBUG;
 84 
 85 //-- public instance meths
 86 
 87 /**
 88  * all parameters will be send to the logger with a LogLevel.DEBUG level value
 89  */
 90 ALogger.prototype.debug = function(){
 91 	if( this.enabled ){		
 92 		var args = new Array( LogLevel.DEBUG );
 93 		for( var i = 0, n = arguments.length ; i < n ; i++ ){
 94 			args.push( arguments[ i ] );
 95 		}
 96 		if( args[ 0 ].ordinal <= this.levelTreshold.ordinal ) this._log.apply( this, args );
 97 	}
 98 }

156 /**
157  * LogCentre
158  * ---------
159  *
160  * outputs to various loggers
161  *
162  * usage: <code>
163  	logger = new LogCentre();
164  	logger.addLogger( new ConsoleLogger() );
165  	logger.addLogger( new MyVerySpecialLogger() );
166  	logger.info( 'Hello World(s)' ); //sends "Hello World(s)" to all loggers
167  </code>
168  * @class LogCentre
169  * @extends ALogger
170  * @author camille.reynders@ibbt.be
171  * @version 1.0.0
172  * @constructor
173  */
174  function LogCentre(){	
175  }
176  
177  //-- prototypal inheritance
178 
179 LogCentre.prototype = new ALogger();
180 LogCentre.prototype.constructor = LogCentre;
181 
182 //-- private instance props
183 
184 /**
185  * @private
186  */
187 LogCentre.prototype._loggers = new Array();
188 
189 //-- public instance meths
190 
191 /**
192  * adds a logger to receive calls and send output
193  * @param {ALogger} logger
194  */
195 LogCentre.prototype.addLogger = function( logger ){
196 	var index = this._loggers.indexOf( logger );
197 	if( index < 0 ){
198 		this._loggers.push( logger );
199 	}
200 }
201 
202 /**
203  * remove a logger to receive calls and send output
204  * @param {ALogger} logger
205  * @return {Boolean} success of removal
206  */
207 LogCentre.prototype.removeLogger = function( logger ){
208 	var index = this._loggers.indexOf( logger );
209 	if( index >= 0 ){
210 		this._loggers.splice( index, 1 );
211 		return true;
212 	}
213 	
214 	return false;
215 }
216 
217 
218 //-- private instance meths
219 
220 /**
221  * override ALogger.prototype._log
222  * @private
223  * @see ALogger#_log
224  */
225 LogCentre.prototype._log = function( level ){
226 	var args = new Array();
227 	for( var i = 0, n = arguments.length ; i < n ; i++ ){
228 		args.push( arguments[ i ] );
229 	}
230 
231 	for( var i = 0, n = this._loggers.length ; i < n ; i++ ){
232 		var logger = this._loggers[ i ];
233 		logger._log.apply( logger, args );
234 	}
235 	
236 }

239 /**
240  * ConsoleLogger
241  * -------------
242  * sends output to the console
243  *
244  * usage: <code>
245  var name = "creynders";
246  logger = new ConsoleLogger();
247  logger.enabled = true; // is true by default
248  logger.separator = " -- "; // is " ## " by default
249  logger.levelTreshold = LogLevel.SYSTEM; //only outputs LogLevel.SYSTEM and lower
250  logger.system( name, "says hi" ); // sends " -- SYSTEM -- creynders says hi" to the console
251  logger.info( "this doesn't pass the treshold" ); // will not be send to the console, since it's higher than the levelTreshold property value
252  </code>
253  * @class ConsoleLogger 
254  * @extends ALogger
255  * @author camille.reynders@ibbt.be
256  * @version 1.0.0
257  * @constructor
258  */
259 function ConsoleLogger(){
260 }
261 
262 //-- prototypal inheritance
263 
264 ConsoleLogger.prototype = new ALogger();
265 ConsoleLogger.prototype.constructor = ConsoleLogger;
266 
267 //-- public instance props
268 ConsoleLogger.prototype.name = 'ConsoleLogger';
269 ConsoleLogger.prototype.separator = " ## ";
270 
271 //-- private instance meths
272 
273 /**
274  * override ALogger.prototype._log
275  * @private
276  * @see ALogger#_log
277  */
278 ConsoleLogger.prototype._log = function( level ){
279 	var args = new Array();
280 	for( var i = 1, n = arguments.length ; i < n ; i++ ){
281 		args.push( arguments[ i ] );
282 	}
283 
284 	console.log( this.separator + level.name.toUpperCase() + this.separator + args.join( ', ' ) );
285 }
