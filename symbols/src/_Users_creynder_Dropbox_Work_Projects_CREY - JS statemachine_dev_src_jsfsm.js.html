 18     /**
 19      * @class
 20      * @constructor
 21      * @param {String} name
 22      * @param {Object} [data]
 23      */
 24     jsfsm.State = function( name, data ){
 25         this.name = name;
 26         /**
 27          * @type Array
 28          * @private
 29          */
 30         this._childrenList = undefined;
 31         /**
 32          * @type Object
 33          * @private
 34          */
 35         this._childIndexesByName = undefined;
 36         /**
 37          * @type jsfsm.State
 38          * @private
 39          */
 40         this._parent = undefined;
 41         this._entryAction = undefined;
 42         this._exitAction = undefined;
 43         this._entryGuard = undefined;
 44         this._exitGuard = undefined;
 45 
 46         if( data ){
 47             this.setParent( data.parent );
 48             this.setEntryAction( data.entryAction );
 49             this.setExitAction( data.exitAction );
 50             this.setEntryGuard( data.entryGuard );
 51             this.setExitGuard( data.exitGuard );
 52             this.setTransitions( data.transitions );
 53         }
 54     };
 55 
 56     jsfsm.State.prototype = {
 57         /**
 58          * @type String
 59          * @default "State"
 60          */
 61         fqn : 'State',
 62 
 63         /**
 64          * @type {String}
 65          * @default undefined
 66          */
 67         name : undefined,
 68 
 69         addChild: function( state ){
 70             if( this._childrenList == undefined ){
 71                 this._childrenList = [];
 72                 this._childIndexesByName = {};
 73             }
 74             var index = this._childIndexesByName.hasOwnProperty( state.name )
 75                 ? this._childIndexesByName[ state.name ]
 76                 : this._childrenList.length;
 77 
 78             this._childrenList[ index ] = state;
 79             this._childIndexesByName[ state.name ] = index;
 80         },
 81 
 82         removeChild : function( stateName ){
 83             if( this._childIndexesByName.hasOwnProperty( stateName ) ){
 84                 var index = this._childIndexesByName[ stateName ];
 85                 delete this._childIndexesByName[ stateName ];
 86                 this._childrenList.splice( index, 1 );
 87             }
 88         },
 89 
 90         setParent : function( state ){
 91             if( this._parent ) this._parent.removeChild( this );
 92             this._parent = state;
 93             if( state ) this._parent.addChild( this );
 94         },
 95         /**
 96          * @return jsfsm.State
 97          */
 98         getParent : function(){
 99             return this._parent;
100         },
101 
102         setEntryAction : function( entryAction ){
103             this._entryAction = entryAction;
104         },
105         getEntryAction : function(){
106             return this._entryAction;
107         },
108 
109         setExitAction : function( exitAction ){
110             this._exitAction = exitAction;
111         },
112         getExitAction : function(){
113             return this._exitAction;
114         },
115 
116         setEntryGuard : function( entryGuard ){
117             this._entryGuard = entryGuard;
118         },
119         getEntryGuard : function(){
120             return this._entryGuard;
121         },
122 
123         setExitGuard : function( exitGuard ){
124             this._exitGuard = exitGuard;
125         },
126         getExitGuard : function(){
127             return this._exitGuard;
128         },
129 
130         /**
131          * @private
132          * @param {Function} guard
133          * @param {jsfsm.State} from
134          * @param {jsfsm.State} to
135          */
136         _executeCallback : function( callback, from, to ){
137             return ( callback == undefined ) || callback.call( this, from, to );
138         },
139 
140         doEntryAction : function( fromState ){
141             return this._executeCallback( this._entryAction, this, toState );
142         },
143 
144         doExitAction : function( toState ){
145             return this._executeCallback( this._exitAction, this, toState );
146         },
147 
148         entryIsAllowed : function( fromState ){
149             return this._executeCallback( this._entryGuard, fromState, this );
150         },
151 
152         exitIsAllowed : function( toState ){
153             return this._doGuard( this._exitGuard, this, toState );
154         },
155 
156         setTransitions : function( transitions ){
157             for( var transitionName in transitions ){
158                 this.addTransition( transitionName, transitions[ transitionName ] );
159             }
160         },
161 
162         addTransition : function( transitionName, from ){
163             if( this._transitions == undefined ) this._transitions = {};
164             this._transitions[ transitionName ]  = from;
165         },
166 
167         removeTransition : function( transitionName ){
168             if( this._transitions.hasOwnProperty( transitionName ) ) delete this._transitions[ transitionName ];
169         },
170 
171         /**
172          * @internal
173          * @param branch
174          */
175         _hasTransitionFor : function( branch ){
176             var state = branch.pop();
177             for( var transitionName in this._transitions ){
178                 /**
179                  * @type {Array|String}
180                  */
181                 if( this._transitions[ transitionName ].indexOf( state.name ) >= 0 ) return true;
182             }
183 
184             return ( branch.length > 0 ) && this._hasTransitionFor( branch );
185         },
186 
187 
188         destroy : function(){
189             this._children = undefined;
190             this._parent = undefined;
191         }
192     };

194     /**
195      * @class
196      * @constructor
197      */
198     jsfsm.StateMachine = function( name ){
199         this._states = {};
200         this._rootState = new jsfsm.State( '*' );
201         this._currentStateBranch = [ this._rootState ];
202     };
203 
204     /**
205      * @type String
206      * @default 'StateMachine'
207      */
208     jsfsm.StateMachine.prototype = {
209         fqn : 'StateMachine',
210 
211         /**
212          * @return {jsfsm.State}
213          */
214         getCurrentState : function(){
215             return this._currentStateBranch[ this._currentStateBranch.length -1 ];
216         },
217 
218         createState : function( stateName, stateData ){
219             var state = new jsfsm.State( stateName, stateData );
220             this.addState( state )
221 
222             return this;
223         },
224 
225         addState : function( state ){
226             this._states[ state.name ] = state;
227 
228             return this;
229         },
230 
231         removeState : function( stateName ){
232             if( this._states.hasOwnProperty( stateName ) ) delete this._states[ stateName ];
233 
234             return this;
235         },
236 
237         _getPathFromRootToState : function( state ){
238             //root will not be included
239             var path = [];
240             do{
241                 path.unshift( state );
242                 state = state.getParent();
243             }while( state != this.root )
244             return path;
245         },
246 
247         _getShortestRoute : function( rootToBegin, rootToEnd ){
248             var i, n = Math.min( rootToBegin.length, rootToEnd.length );
249             for( i=0 ; i<n ; i++ ){
250                 if( rootToBegin[ i ] != rootToEnd[ i ] ) break;
251             }
252 
253             var up = rootToBegin.slice( i ).reverse();
254             var down = rootToEnd.slice( i );
255 
256             return {
257                 up : up,
258                 down : down
259             }
260         },
261 
262         _getInitialChildren:function( node ){
263             var initialState = node.getInitialState();
264             var result = [];
265             while( initialState ){
266                 result.push( initialState );
267                 initialState = initialState.getInitialState();
268             }
269             return result;
270         },
271 
272         changeState : function( stateName ){
273             var newState = this._states[ stateName ];
274             if( newState !== undefined && newState !== this.getCurrentState() && newState._hasTransitionFor( this._currentStateBranch ) ){
275 
276                 var newStatePath = this._getPathFromRootToState( newState );
277                 var streams = this._getShortestRoute( this._currentStateBranch, newStatePath );
278 
279                 var args = [newState]
280                 var proceed = true;
281                 proceed = proceed && this._applyToEachState( streams.up,     jsfsm.State.prototype.exitIsAllowed,    args, true );
282                 proceed = proceed && this._applyToEachState( streams.down,   jsfsm.State.prototype.entryIsAllowed,   args, true );
283                 proceed = proceed && this._applyToEachState( streams.up,     jsfsm.State.prototype.doExitAction,     args, false );
284                 proceed = proceed && this._applyToEachState( streams.down,   jsfsm.State.prototype.doEntryAction,    args, false );
285 
286             }
287         },
288 
289 
290         _applyToEachState : function( statesCollection, callback, args, allowInterrupt ){
291             if( allowInterrupt == undefined ) allowInterrupt = false;
292             var proceed = true;
293             for( var i=0, n=statesCollection.length; ( ! allowInterrupt || proceed ) && i<n ; i++){
294                 var state = statesCollection[ i ];
295                 proceed = callback.apply( state, args );
296             }
297         }
298     }
