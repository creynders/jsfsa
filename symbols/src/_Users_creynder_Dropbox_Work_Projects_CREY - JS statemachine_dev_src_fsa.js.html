 19     /**
 20      * @class
 21      * @constructor
 22      * @param {String} name
 23      * @param {Object} [data]
 24      */
 25     fsa.State = function( name, data ){
 26         /**
 27         * @type String
 28         * @default "State"
 29         * @const
 30         */
 31         this.FQN = 'State';
 32 
 33         /**
 34         * @type String
 35         */
 36         this.name = '';
 37 
 38         /**
 39         * name of the parent state
 40         * @type String
 41         */
 42         this.parent = '';
 43 
 44         /**
 45         * @type Boolean
 46         */
 47         this.isInitial = false;
 48 
 49         /**
 50         * @private
 51         * @type Object
 52         */
 53         this._actions = {};
 54 
 55         /**
 56         * @private
 57         * @type Object
 58         */
 59         this._guards = {};
 60 
 61         /**
 62         * @private
 63         * @type {Object}
 64         */
 65         this._transitions = {};
 66 
 67         if( data ){
 68             if( data.name ){
 69                 name = data.name;
 70             }
 71             if( data.isInitial ) {
 72                 this.isInitial = true;
 73             }
 74             if( data.guards ){
 75                 if( data.guards.enter ) this._addCallbacks( this._guards, 'enter', data.guards.enter );
 76                 if( data.guards.exit ) this._addCallbacks( this._guards, 'exit', data.guards.exit );
 77             }
 78             if( data.actions ){
 79                 if( data.actions.enter ) this._addCallbacks( this._actions, 'enter', data.actions.enter );
 80                 if( data.actions.exit ) this._addCallbacks( this._actions, 'exit', data.actions.exit );
 81             }
 82             if( data.parent ){
 83                 this.parent = data.parent;
 84             }
 85             if( data.transitions ){
 86                 this._addTransitions( data.transitions );
 87             }
 88 
 89             this._addTransitions( data, fsa.State._configMembers );
 90         }
 91         this._parseName( name );
 92     };

244         /**
245          * @internal
246          * @param {String} eventName
247          * @param {fsa.State} fromState
248          * @param {fsa.State} toState
249          */
250         _executeActions : function( eventName, fromState, toState ){
251             if( this._actions.hasOwnProperty( eventName ) ){
252                 var payload = {
253                     event : eventName,
254                     from : fromState.name,
255                     to : toState.name
256                 }
257                 for( var i=0, n=this._actions[ eventName ].length ; i<n ; i++ ){
258                     var callback = this._actions[ eventName ][ i ];
259                     callback.call( this, payload );
260                 }
261             }
262         },

264         /**
265          * @internal
266          * @param {String} eventName
267          * @param {fsa.State} fromState
268          * @param {fsa.State} toState
269          * @return {Boolean}
270          */
271         _executeGuards : function( eventName, fromState, toState ){
272             if( this._guards.hasOwnProperty( eventName ) ){
273                 var payload = {
274                     event : eventName,
275                     from : fromState.name,
276                     to : toState.name
277                 };
278                 for( var i=0, n=this._guards[ eventName ].length ; i<n ; i++ ){
279                     var callback = this._guards[ eventName ][ i ];
280                     if( ! callback.call( this, payload ) ) return false;
281                 }
282             }
283 
284             return true;
285         },

287         /**

289          * @param {Array} callbacksList
290          * @param {String} eventName
291          * @param {Function} callback
292          */
293         _getCallbackIndex : function( callbacksList, eventName, callback ){
294             if( callbacksList && callbacksList.hasOwnProperty( eventName ) ){
295                 return callbacksList[ eventName ].indexOf( callback );
296             }
297 
298             return -1;
299         },

302         /**
303          * @private
304          * @param {Function[]} receiver
305          * @param {String} eventName
306          * @param {Function[]|Function} callbacks
307          */
308         _addCallbacks: function( receiver, eventName, callbacks ){
309             if( typeof callbacks == "function" ){
310                 this._addCallback( receiver, eventName, callbacks );
311             }else{
312                 for( var i=0, n=callbacks.length ; i<n ; i++ ){
313                     this._addCallback( receiver, eventName, callbacks[ i ] );
314                 }
315             }
316         },

319         /**
320          * @private
321          * @param {Function[]} receiver
322          * @param {String} eventName
323          * @param {Function} callback
324          */
325         _addCallback : function( receiver, eventName, callback ){
326             if( ! receiver.hasOwnProperty( eventName ) ) receiver[ eventName ] = [];
327             if( receiver[ eventName ].indexOf( callback ) < 0 ) receiver[ eventName ].push( callback );
328         },

330         /**
331          * @private
332          * @param {Function[]} callbacksList
333          * @param {String} eventName
334          * @param {Function} callback
335          */
336         _removeCallback : function( callbacksList, eventName, callback ){
337             var index = this._getCallbackIndex( callbacksList, eventName, callback );
338             if( index >= 0 ) callbacksList[ eventName ].splice( index, 1 );
339         },

341         /**
342          * @private
343          * @param {Object} data
344          * @param {String[]} [skip]
345          */
346         _addTransitions : function( data, skip ){
347             for( var transitionName in data ){
348                 if( skip && skip.indexOf( transitionName ) >= 0 ) continue;
349                 this.addTransition( transitionName, data[ transitionName ] );
350             }

369     fsa._Node.prototype = {
370         /**
371          *
372          * @param {fsa._Node} node
373          */
374         addChild : function( node ){
375             if( ! this.children ) this.children = {};
376             node.parent = this;
377             if( node.state.isInitial ) this.initialChild = node;
378             this.children[ node.state.name ] = node;
379         },
380         /**
381          *
382          * @param {fsa._Node} node
383          */
384         removeChild : function( node ){
385             if( this.initialChild === node ) this.initialChild = undefined;
386             delete this.children[ node.state.name ];
387         },
388         destroy : function(){
389             for( var stateName in this.children ){
390                 this.children[ stateName ].destroy();
391             }
392             this.state = undefined;
393             this.parent = undefined;
394             this.children = undefined;
395         },
396         /**
397          * @return {fsa._Node[]}
398          */
399         getInitialBranch : function(){
400             var result = [];
401             var initial = this.initialChild;
402             while( initial ){
403                 result.push( initial );
404                 initial = initial.initialChild;
405             }
406 
407             return result;
408         }

415     fsa.Automaton = function( data ){
416         this._nodes = {};
417         this._rootNode = new fsa._Node( new fsa.State( '*' ), true );
418         this._currentBranch = [ this._rootNode ];
419         if( data ){
420             this.parse( data );
421         }

424     fsa.Automaton.prototype = {
425         /**
426          * @const
427          * @type String
428          * @default 'Automaton'
429          */
430         FQN : 'Automaton',

432         /**
433          * @return {fsa.State}
434          */
435         getRootState : function(){
436             return this._rootNode.state;
437         },

439         /**

442         getCurrentState : function(){
443             return this._currentBranch[ this._currentBranch.length -1 ].state;
444         },

446         /**
447          *
448          * @param {String} stateName
449          * @param {Object} stateData
450          * @return {fsa.Automaton} the instance of {@link fsa.Automaton} that is acted upon
451          */
452         createState : function( stateName, stateData ){
453             var state = new fsa.State( stateName, stateData );
454             this.addState( state );

459         /**
460          *
461          * @param {fsa.State} state
462          * @return {fsa.Automaton} the instance of {@link fsa.Automaton} that is acted upon
463          */
464         addState : function( state ){
465             if( ! this.hasState( state.name ) ){
466                 var node = new fsa._Node( state );
467                 var parentNode = ( state.parent ) ? this._nodes[ state.parent ] : this._rootNode;
468                 parentNode.addChild( node );
469                 if( state.isInitial && parentNode.state == this.getCurrentState() ) this._currentBranch.push( node );
470                 this._nodes[ state.name ] = node;
471             }
472 
473             return this;
474         },
475 
476         /**
477          *
478          * @param {String} stateName
479          * @return {Boolean}
480          */
481         hasState : function( stateName ){
482             return this._nodes.hasOwnProperty( stateName );
483         },
484 
485         /**
486          *
487          * @param {String} stateName
488          * @return {fsa.State} <code>undefined</code> if no state with name <code>stateName</code> was found.
489          */
490         getState : function( stateName ){
491             if( this.hasState( stateName ) ) return this._nodes[ stateName ].state;
492 
493             return undefined;
494         },
495 
496         /**
497          *
498          * @param {String} stateName
499          * @return {fsa.Automaton} the instance of {@link fsa.Automaton} that is acted upon
500          */
501         removeState : function( stateName ){
502             if( this.hasState( stateName ) ){
503                 var node = this._nodes[ stateName ];
504                 var parentNode = node.parent;
505                 parentNode.removeChild( node );
506                 var index = this._currentBranch.indexOf( node );
507                 if( index >= 0 ) this._currentBranch.splice( index, this._currentBranch.length - index );
508                 node.destroy();
509                 delete this._nodes[ stateName ];
510             }
511 
512             return this;
513         },

515         /**
516          *
517          * @param {String} transitionName
518          * @return {fsa.Automaton} the instance of {@link fsa.Automaton} that is acted upon
519          */
520         doTransition : function( transitionName ){
521             var runner;
522             for( var i=this._currentBranch.length -1, n = 0  ; i>=n ; i-- ){
523                 runner = this._currentBranch[ i ].state
524                 if( runner.hasTransition( transitionName ) ) break;
525             }
526             if( i>=0 ){
527                 //transition found somewhere in the _currentStateBranch
528                 var node = this._nodes[ runner.getTransition( transitionName ) ];
529                 if( node ){
530                     //TODO: determine what to do if node not found?? Currently failing silenlty
531 
532                     var newStateBranch = this._getFullBranch( node );
533                     var streams = this._getShortestRoute( this._currentBranch, newStateBranch );
534 
535                     var args = [ this.getCurrentState(), node.state ];
536                     var proceed = this._applyToEachNode( streams.up,     fsa.State.prototype._executeGuards,    [ 'exit' ].concat( args ), true );
537                     var initialNodes;
538                     if( proceed ){
539                         initialNodes = node.getInitialBranch();
540                         proceed = this._applyToEachNode( streams.down.concat( initialNodes ),   fsa.State.prototype._executeGuards,   [ 'enter' ].concat( args ), true );
541                     }
542                     if( proceed ) {
543                         this._applyToEachNode( streams.up,     fsa.State.prototype._executeActions,     [ 'exit' ].concat( args ), false );
544                         this._applyToEachNode( streams.down,   fsa.State.prototype._executeActions,    [ 'enter' ].concat( args ), false );
545                         this._currentBranch = newStateBranch.concat( initialNodes );
546                     }
547                 }
548             }
549         },

571         /**
572          * @private
573          * @param {fsa._Node} node
574          * @return {fsa._Node[]}
575          */
576         _getFullBranch : function( node ){
577             var branch = [];
578             do{
579                 branch.unshift( node );
580                 node = node.parent;
581             }while( node );
582 
583             return branch;
584         },
585 
586 
587         /**
588          * @private
589          * @param {fsa._Node[]} rootToBegin
590          * @param {fsa._node[]} rootToEnd
591          * @return {Object}
592          */
593         _getShortestRoute : function( rootToBegin, rootToEnd ){
594             var i, n = Math.min( rootToBegin.length, rootToEnd.length );
595             for( i=0 ; i<n ; i++ ){
596                 if( rootToBegin[ i ] != rootToEnd[ i ] ) break;
597             }
598 
599             var up = rootToBegin.slice( i ).reverse();
600             var down = rootToEnd.slice( i );
601 
602             return {
603                 up : up,
604                 down : down
605             }
606         },
607 
608         /**
609          * @private
610          * @param {fsa._Node[]} nodesList
611          * @param {Function} callback
612          * @param {Array} args
613          * @param {Boolean} [allowInterrupt=false]
614          */
615         _applyToEachNode : function( nodesList, callback, args, allowInterrupt ){
616             if( allowInterrupt == undefined ) allowInterrupt = false;
617             var proceed = true;
618             for( var i=0, n=nodesList.length; ( ! allowInterrupt || proceed ) && i<n ; i++){
619                 var state = nodesList[ i ].state;
620                 proceed = callback.apply( state, args );
621             }
622 
623             return proceed
624         }
